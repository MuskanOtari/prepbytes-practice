Question 1: Find the number of digits

Solution Code:

          const Find_Digits = (N) => 
          {
           
           let count = 0;
           while(N!=0)
           {
             count++;
             N= Math.floor(N/10);
           }
           
          return count;
             
          };

Question 2: Find the Fives.

Solution Code:

          const Find_Five = (n) => {
          let count=0;
          while(n!==0)
          {
            let lastDigit =n%10;
            if(lastDigit==5)
            {
              count++;
            }
            n=Math.floor(n/10);
            
          }
          return count;
            
          };

Question 6: Print the Pattern.

Solution Code:

            let s ="";

          };

Question 7: Check whether a Number is a prime or not.

Solution Code:

          const Prime_Check = (N) => 
          {
             
             let count = 0;
            for(let i =1; i<=N; i++){
              if(N%i==0)
              {
                count++;
              }
              
            }
            if(count ==2)
            {
                return "YES";
            }
            else
            {
              return "NO";
            }
            
          };
