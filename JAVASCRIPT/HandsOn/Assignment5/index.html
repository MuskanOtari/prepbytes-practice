Question 1: Count Characters.

Solution Code:

          const countCharacters = (S) => 
          {
              //we have to find the occurance of capital A
              //and capital D in string
              
              let countOfA=0, countOfD=0;
              for(let i=0; i< S.length;i++)
              {
                if(S[i]=="A")
                  countOfA++;
                
                 if(S[i]=="D")
                  countOfD++;
              }
              //Array Declaration
             let array=[];
              array[0]=countOfA;
              array[1]=countOfD;
          
              return array;
              
          };

Question 6: Find the Winner

Solution Code:

 
          var Game_Winner = (S) => 
          {
            let countMaxOfA=0;
            let countMaxOfD=0;
            for(let i=0 ; i< S.length;i++){
              if(S[i]=="A"){
              countMaxOfA++;
            }
            else if(S[i]=="D"){
              countMaxOfD++;
            }
            }
            if(countMaxOfA>countMaxOfD){
              return "Aditya";
            }
            else if(countMaxOfA< countMaxOfD){
              return "Danish";
            }
            else return "Draw";
          }
