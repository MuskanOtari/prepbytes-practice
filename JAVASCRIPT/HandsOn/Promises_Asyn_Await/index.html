Question 3: Write promise function to print numbers from 1 to 7, in which 1 should be printed after 1 sec , 2 should be printed after 2 sec, 3 should be printed after 3 sec and so on. 

          Numbers
          1
          2
          3
          4
          5
          6
          7

Output Code:

 
          function print(num) {
          return new Promise((resolve) => {
              setTimeout(() => {
                  console.log(num);
                  resolve()
              }, 1000)
          })
      }
      let demo = Promise.resolve();
      for (let i = 1; i <= 7; i++) {
          demo = demo.then(() => print(i));
      }

Question 4: Create a promise function accepting a argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed then it should go to reject state and catch the error and print Promise Rejected.

Output Code:

          function print(a){
            return new Promise(function (resolve, reject){
             if(a==="yes"){
              resolve("Promise is Resolved");
             }
             else{
             reject("Promise is Rejected");
             }
            })
        }
        print("yes")
            .then((res)=>console.log(res))
            .catch((err)=>console.log(err))

Question 6: Create examples to explain callback hell function. 

          function myFunc1(callback) {
            console.log("Function 1")
            callback();
        }
        function myFunc2(callback) {
            console.log("Function 2")
            callback();
        }
        function myFunc3(callback) {
            console.log("Function 3")
            callback();
        }
        function myFunc4() {
            console.log("Function 4")
        }
    
        myFunc1(function () {
            myFunc2(function () {
                myFunc3(function () {
                    myFunc4();
                })
            })
        })
