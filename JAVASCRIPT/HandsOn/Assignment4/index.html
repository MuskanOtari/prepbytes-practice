Question 8: Unique Color Shirt

Solution Code:

          function Unique_Shirts (arr,N) {
          
           let uniqueElement =0;
          //Samecolorshirt>=2{}
          for(let j=0 ; j< arr.length; j++){
            let count =0;
            for(let i=0; i< N; i++){
              if(arr[j]==arr[i])
              {
                count++;
              }
            }
              if(count==1){
                uniqueElement++;
              }
          }
          return uniqueElement;
           
          }

Question 9:Min and Max

Solution Code:

 
          function arrayMin(arr) {
         let min=arr[0];
         for(let i=1; i< arr.length; i++)
         {
           if(min>arr[i]){
             min=arr[i];
           }
           
         }  
         return min;
         }
         
         function arrayMax(arr) {
          
         let max=arr[0];
         for(let i=1; i< arr.length; i++)
         {
           if(max< arr[i]){
             max=arr[i];
           }
           
         }
         return max;
         }
